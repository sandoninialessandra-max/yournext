CREATE TABLE IF NOT EXISTS user_cities (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
  city_name TEXT NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(user_id, city_name)
);

CREATE TABLE IF NOT EXISTS visited_restaurants (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
  restaurant_id TEXT NOT NULL,
  restaurant_name TEXT NOT NULL,
  restaurant_cover TEXT,
  restaurant_address TEXT,
  restaurant_city TEXT,
  restaurant_cuisine TEXT,
  restaurant_price_level INTEGER,
  status TEXT DEFAULT 'visited' CHECK (status IN ('visited', 'wishlist')),
  rating NUMERIC(2,1) CHECK (rating >= 0.5 AND rating <= 5),
  is_favorite BOOLEAN DEFAULT FALSE,
  notes TEXT,
  labels TEXT[] DEFAULT '{}',
  created_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(user_id, restaurant_id)
);

CREATE TABLE IF NOT EXISTS restaurant_suggestions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  from_user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
  to_user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
  restaurant_id TEXT NOT NULL,
  restaurant_name TEXT NOT NULL,
  restaurant_cover TEXT,
  restaurant_city TEXT,
  restaurant_cuisine TEXT,
  comment TEXT DEFAULT '',
  read BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

ALTER TABLE user_cities ENABLE ROW LEVEL SECURITY;
ALTER TABLE visited_restaurants ENABLE ROW LEVEL SECURITY;
ALTER TABLE restaurant_suggestions ENABLE ROW LEVEL SECURITY;

-- user_cities: owner-only
CREATE POLICY "user_cities_select" ON user_cities FOR SELECT TO authenticated USING (auth.uid() = user_id);
CREATE POLICY "user_cities_insert" ON user_cities FOR INSERT TO authenticated WITH CHECK (auth.uid() = user_id);
CREATE POLICY "user_cities_update" ON user_cities FOR UPDATE TO authenticated USING (auth.uid() = user_id);
CREATE POLICY "user_cities_delete" ON user_cities FOR DELETE TO authenticated USING (auth.uid() = user_id);

-- visited_restaurants: owner-only
CREATE POLICY "visited_restaurants_select" ON visited_restaurants FOR SELECT TO authenticated USING (auth.uid() = user_id);
CREATE POLICY "visited_restaurants_insert" ON visited_restaurants FOR INSERT TO authenticated WITH CHECK (auth.uid() = user_id);
CREATE POLICY "visited_restaurants_update" ON visited_restaurants FOR UPDATE TO authenticated USING (auth.uid() = user_id);
CREATE POLICY "visited_restaurants_delete" ON visited_restaurants FOR DELETE TO authenticated USING (auth.uid() = user_id);

-- restaurant_suggestions: from/to entrambi possono leggere, solo mittente inserisce, solo destinatario aggiorna (per mark-as-read)
CREATE POLICY "restaurant_suggestions_select" ON restaurant_suggestions FOR SELECT TO authenticated
  USING (auth.uid() = from_user_id OR auth.uid() = to_user_id);
CREATE POLICY "restaurant_suggestions_insert" ON restaurant_suggestions FOR INSERT TO authenticated
  WITH CHECK (auth.uid() = from_user_id);
CREATE POLICY "restaurant_suggestions_update" ON restaurant_suggestions FOR UPDATE TO authenticated
  USING (auth.uid() = to_user_id);
```

const FSQ_KEY = import.meta.env.VITE_FOURSQUARE_API_KEY
const FSQ_BASE = 'https://api.foursquare.com/v3/places'

function ensureKey() {
  if (!FSQ_KEY) {
    throw new Error('VITE_FOURSQUARE_API_KEY mancante — configura .env.local con la chiave Foursquare')
  }
}

function formatPlace(item) {
  if (!item) return null
  const loc = item.location || {}
  const name = item.name || 'Locale sconosciuto'
  const address = loc.address || ''
  const photo = item.photos?.[0]
  return {
    id: item.fsq_id,
    name,
    address,
    city: loc.locality || '',
    cuisine: item.categories?.[0]?.name || '',
    priceLevel: item.price ?? null,
    rating: item.rating ?? null,
    cover: photo ? `${photo.prefix}300x300${photo.suffix}` : null,
    mapsUrl: `https://maps.google.com/?q=${encodeURIComponent(`${name} ${address}`.trim())}`,
  }
}

const FIELDS_SEARCH = 'fsq_id,name,location,categories,price,rating,photos'
const FIELDS_DETAIL = 'fsq_id,name,location,categories,price,rating,photos,description,hours,website'

export const foursquare = {
  async search(query, city) {
    ensureKey()
    const params = new URLSearchParams({
      query,
      limit: '12',
      fields: FIELDS_SEARCH,
    })
    if (city) params.set('near', city)
    const res = await fetch(`${FSQ_BASE}/search?${params.toString()}`, {
      headers: { Accept: 'application/json', Authorization: FSQ_KEY },
    })
    const data = await res.json()
    return (data.results || []).map(formatPlace).filter(Boolean)
  },
  async getPlace(id) {
    ensureKey()
    const res = await fetch(`${FSQ_BASE}/${id}?fields=${FIELDS_DETAIL}`, {
      headers: { Accept: 'application/json', Authorization: FSQ_KEY },
    })
    return formatPlace(await res.json())
  },
  async getPopular(city, category) {
    ensureKey()
    const params = new URLSearchParams({
      query: category || '',
      limit: '12',
      sort: 'POPULARITY',
      fields: FIELDS_SEARCH,
    })
    if (city) params.set('near', city)
    const res = await fetch(`${FSQ_BASE}/search?${params.toString()}`, {
      headers: { Accept: 'application/json', Authorization: FSQ_KEY },
    })
    const data = await res.json()
    return (data.results || []).map(formatPlace).filter(Boolean)
  },
  coverUrl(photo) {
    return photo || null
  },
}
```

Note:
- `URLSearchParams` gestisce encoding automatico di query/city/category (sanitizza input contro injection in URL — mitigazione T-02-01).
- Fail-visible: `ensureKey()` è invocata come primo statement di ogni metodo pubblico (`search`, `getPlace`, `getPopular`). Se `VITE_FOURSQUARE_API_KEY` è assente, lancia un `Error` con messaggio italiano esplicito (`VITE_FOURSQUARE_API_KEY mancante — configura .env.local con la chiave Foursquare`). Questo soddisfa il success criterion #6 di ROADMAP §Phase 2 ("l'app fallisce in modo visibile, non silenziosa, se la chiave manca") — rejection nel runtime browser, non lista vuota camuffata da 401.
- NON aggiungere `coverUrl` come helper complesso: è pass-through perché Foursquare già restituisce URL completi dopo il normalizer (D-09).

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser SPA → Foursquare API | Query + city parameter provenienti da input utente attraversano URL |
| Browser SPA → Supabase RPC | Tutte le CRUD passano attraverso RLS applicate lato DB |
| Browser SPA → Gemini/Groq API | Prompt contiene campi di catalogo owned dall'utente (dopo RLS) |
| Developer machine → Supabase SQL editor | DDL applicato manualmente via dashboard (step `[BLOCKING]` Task 2) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Injection | `foursquare.search(query, city)` e `getPopular(city, category)` | mitigate | `URLSearchParams` encodes query/city/category automaticamente (Task 3). No string concatenation nella costruzione URL. |
| T-02-02 | Injection | AI prompt in `ai.getSimilarRestaurants` e `ai.getPersonalizedRestaurantSuggestions` | mitigate | Interpolare SOLO campi catalogo (restaurant_name, restaurant_city, restaurant_cuisine, labels, rating) nel prompt; non interpolare mai `r.notes` o `r.comment` (Task 6). Le notes sono user-input libero con rischio di prompt-injection triviale; restano fuori dal prompt. |
| T-02-03 | Info Disclosure | RLS su `visited_restaurants` / `user_cities` | mitigate | Policy `USING (auth.uid() = user_id)` su SELECT/INSERT/UPDATE/DELETE per entrambe le tabelle (Task 1). Identico al pattern `watched_movies`/`read_books`. |
| T-02-04 | Info Disclosure | RLS su `restaurant_suggestions` (cross-user) | mitigate | SELECT USING `from_user_id OR to_user_id` (entrambe le parti vedono il messaggio); INSERT WITH CHECK `from_user_id = auth.uid()` (impossibile forgere mittente); UPDATE USING `to_user_id = auth.uid()` (solo destinatario può mark-as-read). Pattern identico a `movie_suggestions` (Task 1). |
| T-02-05 | Tampering | Rating / price_level / status data corruption da client malizioso | mitigate | DB-level CHECK constraints: `status IN ('visited', 'wishlist')`, `rating NUMERIC(2,1) CHECK (rating >= 0.5 AND rating <= 5)`. `price_level INTEGER` senza range check al DB ma campo è read-only dal provider (non user-editable). Extra UI-level cap in Plan 02. (Task 1) |
| T-02-06 | Spoofing | `restaurant_suggestions.from_user_id` potrebbe essere forgiato se RLS mancasse | mitigate | RLS INSERT WITH CHECK `from_user_id = auth.uid()` — Postgres rifiuta l'INSERT se il claim JWT non combacia. (Task 1) |
| T-02-07 | Info Disclosure | `VITE_FOURSQUARE_API_KEY` esposta nel bundle SPA | accept | **Accepted trade-off** per app privata invite-only (5-10 utenti), coerente con la decisione già logged in `.planning/codebase/CONCERNS.md` e in PROJECT.md §Out of Scope ("Proxy serverless per chiavi LLM"). Monitoring manuale del consumo. NON proporre proxy in questa fase. |
| T-02-08 | Denial of Service | Un utente esaurisce la quota Foursquare free tier (50 calls/minute) con search spam | accept | Free tier sufficiente per 5-10 utenti. In caso di rate limit, Foursquare risponde 429 e l'app mostra lista vuota — fail naturale. Monitoring manuale. |
| T-02-09 | Denial of Service | `ai.getPersonalizedRestaurantSuggestions` con 500+ visited rows consuma context | mitigate | Cap a 15 recent + 10 favorites + 10 cuisines/cities/labels (via `.slice(0, N)`) nel Task 6. Prompt bounded. |
| T-02-10 | Repudiation | Mancanza di audit log su chi ha inviato suggestion a chi | accept | `restaurant_suggestions` contiene `from_user_id`, `to_user_id`, `created_at` — audit trail implicito coerente con pattern esistente `movie_suggestions`. Sufficiente per app privata. |
| T-02-11 | Elevation of Privilege | Un utente tenta di leggere `user_cities` di un altro utente | mitigate | RLS `USING (auth.uid() = user_id)` su SELECT (Task 1). Impossibile bypassare via SPA senza service_role key (che non è esposta). |
